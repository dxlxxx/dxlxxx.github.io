# 防止通过url直接跳转到home

使用路由导航守卫控制访问权限

router.beforeEach((to, from, next) => {
  // 跳转login直接放行
  if (to.path === '/login') return next()
  // 读取token值
  const tokenStr = window.sessionStorage.getItem('token')
  // 跳转其他页判断token值 不存在直接跳转登录页
  if (!tokenStr) return next('/login')
  // 存在放行
  next()
})
